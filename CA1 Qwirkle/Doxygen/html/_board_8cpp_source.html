400bool Board::isValidPlacement(sf::Vector2i gridPosition, int currentPlayer)
401{
402 if (startingGame) // Only happens once
403 {
404 if (gridPosition.x == col / 2 && gridPosition.y == row / 2)
405 {
406 startingGame = false;
407 return true;
408 }
409 return false;
410 }
411 else
412 {
413 if (currentPlayer == 1)
414 {
415 currentTile = player1Hands[selectedTileIndex];
416 }
417 else
418 {
419 currentTile = player2Hands[selectedTileIndex];
420 }
421
422 bool leftLine = checkLine(gridPosition, currentTile, sf::Vector2i(-1, 0));
423 bool rightLine = checkLine(gridPosition, currentTile, sf::Vector2i(1, 0));
424 bool upLine = checkLine(gridPosition, currentTile, sf::Vector2i(0, -1));
425 bool downLine = checkLine(gridPosition, currentTile, sf::Vector2i(0, 1));
426
427 // Check adjacent lines in all directions
428 bool adjacentLinesValid = checkAdjacentLines(gridPosition, currentTile);
429
430 if (adjacentLinesValid && (leftLine && rightLine && upLine && downLine))
431 {
432 std::cout << "\nValid line formed" << std::endl;
433 return true;
434 }
435 return false;
436 }

444bool Board::checkAdjacentLines(sf::Vector2i gridPosition, const Tile& currentTile)
445{
446 sf::Vector2i adjacentPositions[] =
447 {
448 {gridPosition.x, gridPosition.y - 1}, // above
449 {gridPosition.x, gridPosition.y + 1}, // below
450 {gridPosition.x - 1, gridPosition.y}, // left
451 {gridPosition.x + 1, gridPosition.y} // right
452 };
453
454 for (const auto& position : adjacentPositions)
455 {
456 // Check if the position is within the board boundaries
457 if (position.x >= 0 && position.x < col && position.y >= 0 && position.y < row)
458 {
459 for (const auto& tile : tilesOnBoard)
460 {
461 sf::Vector2i tilePosition = tile.getPosition();
462 if (position == tilePosition)
463 {
464 // Compare color or shape of the current tile with the adjacent tile
465 if (currentTile.getColor() == tile.getColor())
466 {
467 std::cout << "Adjacent tile shares color" << std::endl;
468 return true;
469 }
470 else if (currentTile.getShape() == tile.getShape())
471 {
472 std::cout << "Adjacent tile shares shape" << std::endl;
473 return true;

475 }
476 }
477 }
478 }

489bool Board::checkLine(const sf::Vector2i& gridPosition, const Tile& currentTile, const sf::Vector2i& direction)
490{
491 int countColor = 0;
492 int countShape = 0;
493 bool lineDetected = false;
494
495 std::vector<Tile> lineTiles;
496
497 for (int i = 1; i <= 5; ++i)
498 {
499 int x = gridPosition.x + i * direction.x;
500 int y = gridPosition.y + i * direction.y;
501
502 if (x < 0 || x >= col || y < 0 || y >= row)
503 {
504 break; // reached board boundary
505 }
506
507 bool tileFound = false;
508 for (const auto& tile : tilesOnBoard)
509 {
510 sf::Vector2i tilePosition = tile.getPosition();
511 if (tilePosition.x == x && tilePosition.y == y)
512 {
513 lineTiles.push_back(tile);
514 tileFound = true;
515 }
516 }
517
518 if (!tileFound)
519 {
520 break; // break if a gap in the line is found
521 }
522 }
523
524 // Check if the line exists and the current tile can be placed in it
525 if (!lineTiles.empty())
526 {
527 for (const auto& tile : lineTiles)
528 {
529 if (tile.getColor() == currentTile.getColor())
530 {
531 countColor++;
532 }
533 if (tile.getShape() == currentTile.getShape())
534 {
535 countShape++;
536 }
537 }
538 lineDetected = (countColor == lineTiles.size() || countShape == lineTiles.size());
539 }
540 else
541 {
542 return !lineDetected;
543 }
544 return lineDetected;
545}
